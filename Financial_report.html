from scipy import stats
In [33]:
df_week['weeks_from_start'] = [i for i in range(len(df_week))]
In [34]:
x = df_week['weeks_from_start'].values
y = df_week['revenue'].values/10**3
res = stats.linregress(x, y)
In [35]:
print(f"R-squared: {res.rvalue**2:.2f}")
